# APP 清理缓存

<template>
  <view class="r_all">
    <view class="r_qea" @click="onClear">
      <view>清除缓存</view>
      <view class="uni-flex">
        <view class="current-size">{{ cache.currentSize }}</view>
        <u-icon name="arrow-right" size="14"></u-icon>
      </view>
    </view>
    <view class="r_qea" @click="toAbout">
      <view>关于我们</view>
      <u-icon name="arrow-right" size="14"></u-icon>
    </view>
    <view class="r_qea">
      <view>隐私协议</view>
      <u-icon name="arrow-right" size="14"></u-icon>
    </view>
    <view class="r_qea" @click="onCancel">
      <view>注销账号</view>
      <u-icon name="arrow-right" size="14"></u-icon>
    </view>
    <view class="r_qea" @click="toFeedBack">
      <view>意见反馈</view>
      <u-icon name="arrow-right" size="14"></u-icon>
    </view>
    <view class="r_out" @click="onLoginOut">退出登录</view>
  </view>
</template>

<script>
export default {
  filters: {
    filterSise(value) {},
  },
  data() {
    return {
      cache: null,
    };
  },
  onShow() {
    this.cache = uni.getStorageInfoSync();
    this.cache.currentSize = this.cover(this.cache.currentSize);
    console.log("this.cache", this.cache);
  },
  methods: {
    onCancel() {
      let _this = this;
      uni.showModal({
        content: "您确认要注销账户吗？",
        success(e) {
          if (e.confirm) {
            let postData = {
              state: 2,
            };
            _this.$superApi.getPersonal.upduserinfo(postData).then((res) => {
              if (res.data.code == 0) {
                uni.showToast({
                  title: res.data.msg,
                });
                uni.clearStorageSync();
                let time = setTimeout(() => {
                  clearTimeout(time);
                  uni.redirectTo({
                    url: "../../login/login",
                  });
                }, 1500);
              }
            });
          }
        },
      });
    },
    onLoginOut() {
      uni.clearStorageSync();
      uni.redirectTo({
        url: "../../login/login",
      });
    },
    onClear() {
      let keys = this.cache.keys.filter((item) => {
        if (!["userInfo", "token"].includes(item)) {
          return item;
        }
      });
      console.log("uni.getStorageInfoSync()", uni.getStorageInfoSync());
      let currentSize = this.cache.currentSize;
      uni.showModal({
        content: "确定清除缓存？约" + currentSize,
        success(e) {
          if (e.confirm) {
            keys.map((item, index) => {
              uni.removeStorageSync(item);
              if (index == keys.length - 1) {
                uni.showToast({
                  title: "清除成功",
                });
              }
            });
          }
        },
      });
    },
    cover(limit) {
      var size = "";
      if (limit < 0.1 * 1024) {
        //如果小于0.1KB转化成B
        size = limit.toFixed(2) + "B";
      } else if (limit < 0.1 * 1024 * 1024) {
        //如果小于0.1MB转化成KB
        size = (limit / 1024).toFixed(2) + "KB";
      } else if (limit < 0.1 * 1024 * 1024 * 1024) {
        //如果小于0.1GB转化成MB
        size = (limit / (1024 * 1024)).toFixed(2) + "MB";
      } else {
        //其他转化成GB
        size = (limit / (1024 * 1024 * 1024)).toFixed(2) + "GB";
      }

      var sizestr = size + "";
      var len = sizestr.indexOf("\.");
      var dec = sizestr.substr(len + 1, 2);
      if (dec == "00") {
        //当小数点后为00时 去掉小数部分
        return sizestr.substring(0, len) + sizestr.substr(len + 3, 2);
      }
      return sizestr;
    },
  },
};
</script>

import backOrder from "../components/backOrder.vue";

  components: {
    order,
    backOrder,

    console.log("下拉刷新");

    priceDecimal() {
      return (val) => {
        if (val !== parseInt(val)) return val.slice(-2);
        else return "00";
      };
    },
    // 价格整数
    // priceInt() {
    // 	return val => {

    // 		else return val;
    // 	};
    // }
  },

    toMessage() {
      uni.$emit("toMessage");
    },
    // 未读消息
    listShopMessageByShopId() {
      let postData = {
        shopId: this.vuex_saveShopId,
      };
      this.$Api.listShopMessageByShopId(postData).then((res) => {
        console.log("获取未读消息数量", res);
        if (res.data.code == 0) {
          this.numberOfUnreadMessages = res.data.data.length;
        }
      });
    },
    onPrint() {
      uni.showToast({
        title: "打印机",
        icon: "none",
      });
    },

    // tab到底部获取下一页
    reachBottom: debounce(function() {
      this.loadStatus.splice(this.current, 1, "loading");
      this.selectShopNewOrderList();
    }, 500),
    // 总价
    totalPrice(item) {
      let price = 0;
      item.map((val) => {
        price += parseFloat(val.price);
      });
      return price.toFixed(2);
    },
    // 总件数
    totalNum(item) {
      let num = 0;
      item.map((val) => {
        num += val.number;
      });
      return num;
    },
    // tab栏切换
    change(index) {
      this.swiperCurrent = index;
      this.pageNum = 1;
      this.pageSize = 10;
      console.log("swiperCurrent", index);
      switch (index) {
        case 0:
          this.selStatus = 3;
          break;
        case 1:
          this.selStatus = 4;
          break;
        case 2:
          this.selStatus = 5;
          break;
        case 3:
          this.selStatus = 7;
          break;
      }
      this.orderList = [];
    },

    // 页签动画
    transition({ detail: { dx } }) {
      this.$refs.tabs.setDx(dx);
    },
    animationfinish({ detail: { current } }) {
      this.$refs.tabs.setFinishCurrent(current);
      this.swiperCurrent = current;
      this.current = current;
    },
  },
};
</script>

common.ExportExcel = (data, name = "测试") => {
  return new Promise((resolve, reject) => {
    let wopts = {
      bookType: 'xlsx',
      bookSST: true,
      type: 'binary'
    };
    var workBook = {
      SheetNames: ['Sheet1'],

common.changeData = (s) => {

  //如果存在ArrayBuffer对象(es6) 最好采用该对象
  if (typeof ArrayBuffer !== 'undefined') {

    //1、创建一个字节长度为s.length的内存区域
    let buf = new ArrayBuffer(s.length);

    //2、创建一个指向buf的Unit8视图，开始于字节0，直到缓冲区的末尾
    let view = new Uint8Array(buf);

    //3、返回指定位置的字符的Unicode编码
    for (var i = 0; i != s.length; ++i) view[i] = s.charCodeAt(i) & 0xFF;
    return buf;

  } else {
    let buf = new Array(s.length);
    for (let i = 0; i != s.length; ++i) buf[i] = s.charCodeAt(i) & 0xFF;
    return buf;
  }

<template>
  <vueCropper
    ref="cropper"
    :img="option.img"
    :outputSize="option.size"
    :outputType="option.outputType"
    :info="true"
    :full="option.full"
    :canMove="option.canMove"
    :canMoveBox="option.canMoveBox"
    :original="option.original"
    :autoCrop="option.autoCrop"
    :fixed="option.fixed"
    :fixedNumber="option.fixedNumber"
    :centerBox="option.centerBox"
    :infoTrue="option.infoTrue"
    :fixedBox="option.fixedBox"
  ></vueCropper>
</template>
<script>
import VueCropper from "vue-cropper";
export default {
  data() {
    return {
      option: {
        img: "",
        info: true, // 裁剪框的大小信息
        outputSize: 0.8, // 裁剪生成图片的质量
        outputType: "png", // 裁剪生成图片的格式
        canScale: false, // 图片是否允许滚轮缩放
        autoCrop: true, // 是否默认生成截图框
        fixedBox: false, // 固定截图框大小 不允许改变
        fixed: true, // 是否开启截图框宽高固定比例
        fixedNumber: [7, 5], // 截图框的宽高比例
        full: true, // 是否输出原图比例的截图
        canMoveBox: true, // 截图框能否拖动
        original: true, // 上传图片按照原始比例渲染
        centerBox: false, // 截图框是否被限制在图片里面
        infoTrue: true, // true 为展示真实输出图片宽高 false 展示看到的截图框宽高
      },
    };
  },
  methods: {
    // 保存
    config() {
      var that = this;
      this.$refs.cropper.getCropBlob((data) => {
        let aTime = new Date().getTime(); //取时间戳，给文件命名
        let fileName = aTime + "." + data.type.substr(6); //给文件名加后缀
        let file = new window.File([data], fileName, { type: data.type }); //blob转file
        this.$common.upload("cut" + file.name, file, function(res) {
          // res 裁剪后的图片
          that.item.signHeadImg = res;
        });
      });
    },
    // 反写回的图片需要转换Base64
    setAvatarBase64(src, callback) {
      let _this = this;
      let image = new Image();
      // 处理缓存
      image.src = src + "?v=" + Math.random();
      // 支持跨域图片
      image.crossOrigin = "*";
      image.onload = function() {
        let base64 = _this.transBase64FromImage(image);
        callback && callback(base64);
      };
    },
    transBase64FromImage(image) {
      let canvas = document.createElement("canvas");
      canvas.width = image.width;
      canvas.height = image.height;
      let ctx = canvas.getContext("2d");
      ctx.drawImage(image, 0, 0, image.width, image.height);
      // 可选其他值 image/jpeg
      return canvas.toDataURL("image/png");
    },
  },
};
</script>

let lifeData = {};

try {
  // 尝试获取本地是否存在lifeData变量，第一次启动APP时是不存在的
  lifeData = JSON.parse(localStorage.getItem("lifeData"))
    ? JSON.parse(localStorage.getItem("lifeData"))
    : {};
} catch (e) {}

// 需要永久存储，且下次APP启动需要取出的，在state中的变量名
let saveStateKeys = ["vuex_token", "vuex_sysinfo"];

// 保存变量到本地存储中
const saveLifeData = function(key, value) {
  // 判断变量名是否在需要存储的数组中
  if (saveStateKeys.indexOf(key) != -1) {
    // 获取本地存储的lifeData对象，将变量添加到对象中
    let tmp = JSON.parse(localStorage.getItem("lifeData"));
    // 第一次打开APP，不存在lifeData变量，故放一个{}空对象
    tmp = tmp ? tmp : {};
    tmp[key] = value;

    // 执行这一步后，所有需要存储的变量，都挂载在本地的lifeData对象中
    localStorage.setItem("lifeData", JSON.stringify(tmp));
  }
};

const store = new Vuex.Store({
  state: {
    // 如果上面从本地获取的lifeData对象下有对应的属性，就赋值给state中对应的变量
    // 加上vuex_前缀，是防止变量名冲突，也让人一目了然
    vuex_token: lifeData.vuex_token ? lifeData.vuex_token : null,
    vuex_sysinfo: lifeData.vuex_sysinfo ? lifeData.vuex_sysinfo : null, //系统信息
  },
  mutations: {
    $xStore(state, payload) {
      // 判断是否多层级调用，state中为对象存在的情况，诸如user.info.score = 1

      let nameArr = payload.name.split(".");
      let saveKey = "";
      let len = nameArr.length;
      if (len >= 2) {
        state[nameArr[0]] = state[nameArr[0]] ? state[nameArr[0]] : {};
        let obj = state[nameArr[0]];
        for (let i = 1; i < len - 1; i++) {
          obj = obj[nameArr[i]];
          state[saveKey][i] = payload.value;
        }
        obj[nameArr[len - 1]] = payload.value;
        saveKey = nameArr[0];
      } else {
        // 单层级变量，在state就是一个普通变量的情况
        state[payload.name] = payload.value;
        saveKey = payload.name;
      }

      // 保存变量到本地，见顶部函数定义
      saveLifeData(saveKey, state[saveKey]);
    },
    $TabBar(state, params) {},
  },
  actions: {},
  modules: {},
});

export default store;

# 格式化时间，获取近几天

  • 格式化时间，获取近几天

    function parseTime(time, cDayTo, cFormat) {
  if (arguments.length === 0 || !time) {
    return null;
  }
  const format = cFormat || "{y}-{m}-{d} {h}:{i}:{s}";
  const dayTo = cDayTo || "{1}";

  let date;
  if (typeof time === "object") {
    date = time;
  } else {
    if (typeof time === "string") {
      if (/^[0-9]+$/.test(time)) {
        // support "1548221490638"
        time = parseInt(time);
      } else {
        // support safari
        // https://stackoverflow.com/questions/4310953/invalid-date-in-safari
        time = time.replace(new RegExp(/-/gm), "/");
      }
    }

    if (typeof time === "number" && time.toString().length === 10) {
      time = time * 1000;
    }
    date = new Date(time);
  }
  let reg = /(-?[1-9])+/g;
  date = new Date(date.getTime() + 24 * 60 * 60 * 1000 * dayTo.match(reg));

  const formatObj = {
    y: date.getFullYear(),
    m: date.getMonth() + 1,
    d: date.getDate(),
    h: date.getHours(),
    i: date.getMinutes(),
    s: date.getSeconds(),
    a: date.getDay(),
  };

  let time_str = format.replace(/{(([ymdhisa])+)}/g, (result, key) => {
    const value = formatObj[key];
    console.log(key);
    console.log(value);
    // Note: getDay() returns 0 on Sunday
    if (key === "a") {
      return ["日", "一", "二", "三", "四", "五", "六"][value];
    }
    return value.toString().padStart(2, "0");
  });
  console.log(time_str);
  return time_str;
}
